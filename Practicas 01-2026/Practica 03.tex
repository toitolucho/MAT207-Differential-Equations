\subsubsection*{A. Determine si las siguientes ecuaciones diferenciales son de variable separable, homogénea, lineal o de Bernoulli}

\begin{enumerate}
    \item \( \frac{dy}{dx} = \frac{x^2}{y} \)
    \item \( y' = e^{x-y} \)
    \item \( x y' = (1+x^2)y \)
    \item \( (x^2 - y^2)dx + 2xy\,dy = 0 \)
    \item \( y' = \frac{y+\sqrt{x^2+y^2}}{x} \)
    \item \( \frac{dy}{dx} = \frac{y}{x} - \tan\left(\frac{y}{x}\right) \)
    \item \( \frac{dy}{dx} - 3y = e^x \)
    \item \( x y' + y = x \sin x \)
    \item \( \frac{dr}{d\theta} + r \tan \theta = \sec \theta \)
    \item \( x y' + y = y^2 \ln x \)
    \item \( y' + y = e^x y^3 \)
    \item \( \frac{dy}{dx} - y = y^2 \sin x \)
\end{enumerate}

\subsubsection*{B. Resuelva las siguientes ecuaciones diferenciales Lineales con los valores iniciales dados}

\begin{enumerate}
    \item \textbf{Rpta:} \(y = e^{-x} + \frac{1}{2}(\sin x - \cos x)\) \hfill \(\frac{dy}{dx} + y = \sin x, \quad y(0)= \frac{1}{2}\)
    \item \textbf{Rpta:} \(y = e^{-x^2/2} + 2\) \hfill \(\frac{dy}{dx} + xy = 2x, \quad y(0)=3\)
    \item \textbf{Rpta:} \(y = e^x + (2x+1)\) \hfill \(\frac{dy}{dx} = \frac{2y + (2x-1)e^x}{2x+1}, \quad y(0)=2\)
    \item \textbf{Rpta:} \(y = \sin x + \sin^2 x\) \hfill \(\frac{dy}{dx} - y\cot x = \frac{\sin 2x}{2}, \quad y(\pi/2)=2\)
    \item \textbf{Rpta:} \(y = e^{x+x^2} + e^x\) \hfill \(y' - y = 2x e^{x+x^2}, \quad y(0)=2\)
    \item \textbf{Rpta:} \(y = x^4 + x^3\) \hfill \(x y' - 3y = x^4, \quad y(1)=2\)
    \item \textbf{Rpta:} \(y = \frac{x^4+2x^2+4}{4(x^2+1)}\) \hfill \(\frac{dy}{dx} + \frac{2x}{x^2+1}y = x, \quad y(0)=1\)
    \item \textbf{Rpta:} \(y = x^2+1 + \sqrt{x^2+1}\) \hfill \((x^2+1)y' = x^3 + xy + x, \quad y(0)=2\)
\end{enumerate}

\subsubsection*{C. Resuelva las siguientes ecuaciones diferenciales de Bernoulli con los valores iniciales dados}

\begin{enumerate}
    \item \textbf{Rpta:} \(y(1 + \ln x + x) = 1\) \hfill \(x y' + y = y^2 \ln x, \quad y(1)=1/2\)
    \item \textbf{Rpta:} \(x^{-3}y^{-3} + x^2 = 2\) \hfill \(3 y' + \frac{3}{x}y = 2x^4 y^4, \quad y(1)=1\)
    \item \textbf{Rpta:} \(1 = y e^x (1 - x^2)\) \hfill \(dy + y dx = 2xy^2 e^x dx, \quad y(0)=1\)
    \item \textbf{Rpta:} \(y^3(x+1) = \frac{8}{3}\left[(x+1)^3-1\right]\) \hfill \(3y^2 y' + \frac{y^3}{x+1} - 8(x+1) = 0, \quad y(0)=0\)
    \item \textbf{Rpta:} \(y = \frac{1}{(1+x)\left[1 + \ln|1+x|\right]}\) \hfill \(y' + \frac{y}{x+1} + y^2 = 0, \quad y(0)=1\)
    \item \textbf{Rpta:} \(y = \frac{\tan x + \sec x}{1 + \sin x}\) \hfill \(y - y'\cos x = y^2 \cos x(1-\sin x), \quad y(0)=1\)
    \item \textbf{Rpta:} \(x^2 = \left(1 + e^{x^4-1}\right) y^2\) \hfill \(x y' - y = 2x^2 y(y^2 - x^2), \quad y(1)=\frac{1}{\sqrt{2}}\)
    \item \textbf{Rpta:} \(e^{x^2} y^4 = 2e^{x^2} - 1\) \hfill \(y^3 y' + x y^4 = x e^{-x^2}, \quad y(0)=1\)
\end{enumerate}
